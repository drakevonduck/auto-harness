# Bootstrap Development Harness

You are setting up the development harness for a new project. This harness was extracted from a production app and encodes a full collaboration infrastructure for 2–4 developers working with an AI coding agent. Follow every step precisely. Where you see `{{PLACEHOLDER}}` values, ask me for the correct value before writing the file.

**Required inputs before starting:**
- `{{PROJECT_NAME}}` — npm package name (e.g. `my-app`)
- `{{APP_TITLE}}` — human-readable app name (e.g. `My App`)
- `{{APP_ID}}` — Capacitor app ID in reverse-domain notation (e.g. `com.myorg.myapp`) — skip if not building native mobile
- `{{SUPABASE_PROJECT_REF}}` — production Supabase project ref (e.g. `daeqqbtanngipstrowmh`) — skip if not using Supabase
- `{{PRODUCTION_URL}}` — production domain (e.g. `app.example.com`) — skip if unknown yet
- `{{INCLUDE_CAPACITOR}}` — yes/no: native iOS/Android builds via Capacitor
- `{{INCLUDE_SUPABASE}}` — yes/no: Supabase backend
- `{{INCLUDE_MAPBOX}}` — yes/no: Mapbox GL JS map

Ask me for each of these before writing any files.

---

## What this harness provides

- **Next.js 16 + React 19 + TypeScript 5** (App Router, Turbopack, strict mode)
- **Tailwind CSS v4** with PostCSS
- **ESLint v9** flat config
- **Three-environment model**: local Docker → staging → production, with gated migration promotion
- **Supabase** backend pattern (Auth, Postgres, RLS, RPC, local dev via Docker)
- **Serwist PWA** service worker
- **Capacitor** native mobile shells (iOS/Android)
- **GitHub Actions CI** (TypeScript + ESLint on every PR)
- **PR template** enforcing documentation checklists
- **Claude Code harness**: PostToolUse command logging hook, dev journal generator, permission scopes
- **Structured docs**: ADRs, PRDs, risk register, remediation backlog, dev journal
- **Developer workflow SOP**: branching strategy, migration coordination, work-area ownership

```typescript
import { createServerClient } from "@supabase/ssr";
import { cookies } from "next/headers";

export async function createClient() {
  const cookieStore = await cookies();

  return createServerClient(
    process.env.NEXT_PUBLIC_SUPABASE_URL!,
    process.env.NEXT_PUBLIC_SUPABASE_PUBLISHABLE_DEFAULT_KEY!,
    {
      cookies: {
        getAll() {
          return cookieStore.getAll();
        },
        setAll(cookiesToSet) {
          try {
            cookiesToSet.forEach(({ name, value, options }) =>
              cookieStore.set(name, value, options)
            );
          } catch {
            // Ignore: called from Server Component, cookies are read-only
          }
        },
      },
    }
  );
}
```

### `utils/supabase/middleware.ts`
```typescript
import { createServerClient } from "@supabase/ssr";
import { type NextRequest, NextResponse } from "next/server";

export async function updateSession(request: NextRequest) {
  let supabaseResponse = NextResponse.next({ request });

  const supabaseUrl = process.env.NEXT_PUBLIC_SUPABASE_URL;
  const supabaseKey = process.env.NEXT_PUBLIC_SUPABASE_PUBLISHABLE_DEFAULT_KEY;

  // Graceful degradation: if env vars are missing, skip auth middleware
  if (!supabaseUrl || !supabaseKey) {
    return supabaseResponse;
  }

  const supabase = createServerClient(supabaseUrl, supabaseKey, {
    cookies: {
      getAll() {
        return request.cookies.getAll();
      },
      setAll(cookiesToSet) {
        cookiesToSet.forEach(({ name, value }) =>
          request.cookies.set(name, value)
        );
        supabaseResponse = NextResponse.next({ request });
        cookiesToSet.forEach(({ name, value, options }) =>
          supabaseResponse.cookies.set(name, value, options)
        );
      },
    },
  });

  await supabase.auth.getUser();
  return supabaseResponse;
}
```

## Step 5 — Database promotion script (skip if `{{INCLUDE_SUPABASE}}` is no)

### `scripts/db-promote.sh`
```bash
#!/usr/bin/env bash
# scripts/db-promote.sh
#
# Migration promotion workflow: local → staging → production
#
# Usage:
#   ./scripts/db-promote.sh validate              Reset local DB + lint
#   ./scripts/db-promote.sh staging               Validate locally, push to staging
#   ./scripts/db-promote.sh production            Dry-run against production (safe)
#   FORCE=1 ./scripts/db-promote.sh production    Apply migrations to production
#
# Via npm scripts:
#   npm run db:validate
#   npm run db:promote:staging
#   npm run db:promote:production       (dry-run)
#   FORCE=1 npm run db:promote:production
#
# Prerequisites:
#   - Supabase CLI: npx supabase
#   - Local Supabase running for validate/staging: supabase start
#   - Staging credentials in .env.staging (see .env.staging.example)
#   - Production project linked: supabase link --project-ref {{SUPABASE_PROJECT_REF}}

set -euo pipefail

COMMAND="${1:-help}"

# ── Load staging credentials ──────────────────────────────────────────────────
if [ -f ".env.staging" ]; then
  set -a
  while IFS= read -r line; do
    [[ "$line" =~ ^[[:space:]]*# ]] && continue
    [[ -z "${line// }" ]] && continue
    export "$line"
  done < .env.staging
  set +a
fi

# ── Helpers ───────────────────────────────────────────────────────────────────
step() { echo ""; echo "=== $* ==="; }
ok()   { echo ""; echo "✓ $*"; }
fail() { echo ""; echo "✗ $*" >&2; exit 1; }

# ── Commands ──────────────────────────────────────────────────────────────────

cmd_validate() {
  step "1/2: Reset local database (includes seed data)"
  npx supabase db reset || {
    echo "  (db reset exited non-zero — likely transient 502 during container restart)"
    echo "  Waiting 5s for containers to stabilize..."
    sleep 5
  }

  step "2/2: Lint schema against local"
  npx supabase db lint --local --fail-on error

  ok "Local validation passed — safe to promote"
}

cmd_staging() {
  if [ -z "${STAGING_DB_URL:-}" ]; then
    fail "STAGING_DB_URL is not set.\nCopy .env.staging.example → .env.staging and fill in staging credentials."
  fi

  step "1/3: Validate locally first"
  cmd_validate

  step "2/3: Dry-run against staging"
  npx supabase db push --db-url "$STAGING_DB_URL" --dry-run

  step "3/3: Push migrations to staging"
  npx supabase db push --db-url "$STAGING_DB_URL"

  ok "Staging promotion complete"
  echo "    Verify in the Supabase dashboard before promoting to production."
}

cmd_production() {
  if [ "${FORCE:-}" != "1" ]; then
    echo ""
    echo "This is a dry-run. No changes will be applied to production."
    echo ""
    npx supabase db push --dry-run
    echo ""
    echo "─────────────────────────────────────────────────────────────"
    echo "To apply these migrations to production, run:"
    echo "  FORCE=1 ./scripts/db-promote.sh production"
    echo "  FORCE=1 npm run db:promote:production"
    echo ""
    echo "Ensure staging has been validated first:"
    echo "  ./scripts/db-promote.sh staging"
    echo "─────────────────────────────────────────────────────────────"
    exit 0
  fi

  step "1/2: Final dry-run against production"
  npx supabase db push --dry-run

  step "2/2: Applying migrations to production"
  npx supabase db push

  ok "Production promotion complete"
  echo ""
  echo "Next steps:"
  echo "  1. Update docs/database.md — 'Last validated' date and 'Migrations applied' list"
  echo "  2. Confirm in the Supabase dashboard"
  echo "  3. Run dev journal: bash .claude/hooks/generate-journal-entry.sh"
}

cmd_help() {
  echo ""
  echo "Usage: ./scripts/db-promote.sh <command>"
  echo ""
  echo "Commands:"
  echo "  validate     Reset local DB + lint (no remote changes)"
  echo "  staging      Validate locally, then push to staging"
  echo "  production   Dry-run against production; set FORCE=1 to apply"
  echo ""
  echo "Environment:"
  echo "  FORCE=1      Required to apply migrations to production (safety gate)"
  echo "  .env.staging Staging DB credentials (see .env.staging.example)"
  echo ""
}

case "$COMMAND" in
  validate)   cmd_validate ;;
  staging)    cmd_staging ;;
  production) cmd_production ;;
  *)          cmd_help; exit 1 ;;
esac
```

Make it executable: `chmod +x scripts/db-promote.sh`

### `.claude/hooks/log-command.sh`
```bash
#!/usr/bin/env bash
# PostToolUse hook — logs Bash commands with configurable verbosity.
# Log level from .claude/log-level.txt. Output to .claude/command-log.txt.

set -euo pipefail

PROJECT_DIR="$(cd "$(dirname "$0")/../.." && pwd)"
LOG_FILE="$PROJECT_DIR/.claude/command-log.txt"
LEVEL_FILE="$PROJECT_DIR/.claude/log-level.txt"

LEVEL="status"
if [[ -f "$LEVEL_FILE" ]]; then
  LEVEL="$(cat "$LEVEL_FILE" | tr -d '[:space:]')"
fi

EVENT="$(cat)"
TIMESTAMP="$(date '+%Y-%m-%d %H:%M:%S')"
COMMAND="$(echo "$EVENT" | jq -r '.tool_input.command // "unknown"')"

case "$LEVEL" in
  commands)
    echo "[$TIMESTAMP] $ $COMMAND" >> "$LOG_FILE"
    ;;
  status)
    EXIT_CODE="$(echo "$EVENT" | jq -r '.tool_result.exit_code // "?"')"
    echo "[$TIMESTAMP] $ $COMMAND  [exit: $EXIT_CODE]" >> "$LOG_FILE"
    ;;
  full)
    EXIT_CODE="$(echo "$EVENT" | jq -r '.tool_result.exit_code // "?"')"
    OUTPUT="$(echo "$EVENT" | jq -r '.tool_result.stdout // ""' | head -50)"
    echo "[$TIMESTAMP] $ $COMMAND  [exit: $EXIT_CODE]" >> "$LOG_FILE"
    if [[ -n "$OUTPUT" ]]; then
      echo "$OUTPUT" | sed 's/^/  > /' >> "$LOG_FILE"
    fi
    echo "" >> "$LOG_FILE"
    ;;
  *)
    echo "[$TIMESTAMP] $ $COMMAND" >> "$LOG_FILE"
    ;;
esac
```

Make executable: `chmod +x .claude/hooks/log-command.sh`

### `.claude/hooks/generate-journal-entry.sh`
```bash
#!/usr/bin/env bash
# Auto-generates a dev journal entry from git history.
# Usage: bash .claude/hooks/generate-journal-entry.sh [since-date]
#   since-date: ISO date (YYYY-MM-DD). Defaults to last journal entry date or 7 days ago.

set -euo pipefail

JOURNAL_FILE="docs/dev-journal.md"
TODAY=$(date +%Y-%m-%d)

if [[ -n "${1:-}" ]]; then
  SINCE="$1"
else
  LAST_DATE=$(grep -oE '^## [0-9]{4}-[0-9]{2}-[0-9]{2}' "$JOURNAL_FILE" 2>/dev/null | tail -1 | grep -oE '[0-9]{4}-[0-9]{2}-[0-9]{2}' || true)
  if [[ -n "$LAST_DATE" ]]; then
    SINCE="$LAST_DATE"
  else
    SINCE=$(date -v-7d +%Y-%m-%d 2>/dev/null || date -d "7 days ago" +%Y-%m-%d)
  fi
fi

COMMITS=$(git log --oneline --since="$SINCE" --format="- %s" 2>/dev/null || echo "- (no commits found)")
DIFF_STAT=$(git diff --stat "$(git log --since="$SINCE" --format="%H" --reverse 2>/dev/null | head -1)^..HEAD" 2>/dev/null | tail -1 || echo "(unable to compute)")

DECISIONS=""
ADR_FILES=$(git diff --name-only --since="$SINCE" -- "docs/decisions/" 2>/dev/null | grep -v TEMPLATE || true)
if [[ -n "$ADR_FILES" ]]; then
  while IFS= read -r adr; do
    TITLE=$(head -1 "$adr" 2>/dev/null | sed 's/^# //')
    DECISIONS="${DECISIONS}\n- ${TITLE}"
  done <<< "$ADR_FILES"
fi

ISSUES=$(git log --oneline --since="$SINCE" --format="%s" 2>/dev/null | grep -iE '(fix|bug|issue|error|broken|crash)' | sed 's/^/- /' || true)

ENTRY="## ${TODAY} — Session

**Commits since ${SINCE}:**
${COMMITS}

**Files changed:**
- ${DIFF_STAT}"

if [[ -n "$DECISIONS" ]]; then
  ENTRY="${ENTRY}

**Decisions:**$(echo -e "$DECISIONS")"
fi

if [[ -n "$ISSUES" ]]; then
  ENTRY="${ENTRY}

**Issues / Fixes:**
${ISSUES}"
fi

ENTRY="${ENTRY}

**Notes:**
_[Your observations, context, or next-step thinking here]_

---
"

if [[ ! -f "$JOURNAL_FILE" ]]; then
  cat > "$JOURNAL_FILE" << 'HEADER'
# Development Journal

Auto-generated session entries from git history, with space for free-form notes.
Run `bash .claude/hooks/generate-journal-entry.sh` to add an entry.

---
HEADER
fi

echo "" >> "$JOURNAL_FILE"
echo -e "$ENTRY" >> "$JOURNAL_FILE"

echo "Journal entry appended to ${JOURNAL_FILE} (since ${SINCE})"
```

Make executable: `chmod +x .claude/hooks/generate-journal-entry.sh`
